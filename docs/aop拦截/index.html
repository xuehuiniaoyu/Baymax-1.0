Android Baymax框架 - AOP编程

private Hook.AOP_Listener listener;


@Override
protected void onCreate(Bundle savedInstanceState) {
    super.onCreate(savedInstanceState);
    Baymax.single().getSysHook().inset_aop(listener =new Hook.AOP_Listener() {
        @Override
        public void aop_onBeforeIntercept(String name, Method method, Object[] args) {
        Log.i(TAG, "before:"+name+" , "+args);
        }

        @Override
        public void aop_onAfterIntercept(String name, Method method, Object[] args) {
        Log.i(TAG, "after:"+name+" , "+args);
        }
    });

}

@Override
protected void onDestroy() {
    super.onDestroy();
    Manager.single().getHook().pluck_aop(listener);
}
